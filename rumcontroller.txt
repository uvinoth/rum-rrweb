

// Push Event godoc
// @Summary      Push RUM Event Data
// @Description  Ingest Real User Monitoring (RUM) events for a given organization
// @Tags         Real User Monitoring
// @Accept       json
// @Produce      json
// @Param        rum body dto.EventRequest true "Event Data"
// @Success      201  {object}  map[string]interface{} "message: Events ingested successfully"
// @Failure      400  {object}  map[string]interface{} "error: Bad request"
// @Failure      422  {object}  map[string]interface{} "error: Unprocessable Entity"
// @Failure      500  {object}  map[string]interface{} "error: Internal Server Error"
// @Router       /api/v1/consumer/inferenz/rum/events [post]
func (api *inferenzController) RumEvents(ctx *gin.Context) {
	var req dto.EventRequest

	api.logger.Debug("Rum Events data", zap.Any("rum_event_request", req))

	// Bind JSON
	if err := ctx.ShouldBindJSON(&req); err != nil {
		ctx.JSON(http.StatusBadRequest, gin.H{"error": "invalid JSON"})
		return
	}

	api.logger.Debug("Event data", zap.Any("event_request", req))

	var rumEndpoint models.Rum
	err := api.db.Where("org_id = ? AND endpoint_id = ?", req.OrgId, req.EndpointID).First(&rumEndpoint).Error
	if errors.Is(err, gorm.ErrRecordNotFound) {
		ctx.JSON(http.StatusNotFound, exception.ResourceNotFound("Endpoint", fmt.Sprintf("Endpoint '%d' not found", req.EndpointID)))
		return
	} else if err != nil {
		ctx.JSON(http.StatusInternalServerError, exception.InternalServerError(err.Error()))
		return
	}

	// Default session_id
	if req.SessionID == "" {
		req.SessionID = "unknown"
	}

	// Validate events
	if len(req.Events) == 0 {
		ctx.JSON(http.StatusBadRequest, gin.H{"error": "no events provided"})
		return
	}

	// Insert via service
	if err := api.service.CreateEvent(req, req.OrgId); err != nil {
		ctx.JSON(http.StatusInternalServerError, gin.H{"error": "failed to save events"})
		return
	}

	ctx.JSON(http.StatusCreated, gin.H{
		"message":         "events created successfully",
		"events_received": len(req.Events),
	})
}

// @Router       /api/v1/consumer/inferenz/rum/{org}/{sessionId}/events [get]
func (api *inferenzController) GetSessionEvents(ctx *gin.Context) {
	org := ctx.Param("org")
	sessionId := ctx.Param("sessionId")

	orgId, err := strconv.ParseInt(org, 10, 32)
	if err != nil {
		ctx.JSON(http.StatusBadRequest, gin.H{"error": "Invalid Org ID"})
		return
	}

	events, total, err := api.service.GetEventsBySession(int32(orgId), sessionId)
	if err != nil {
		ctx.JSON(http.StatusInternalServerError, gin.H{"error": "failed to fetch error events"})
		return
	}

	ctx.JSON(http.StatusOK, gin.H{
		"events":     events,
		"total":      total,
		"session_id": sessionId,
	})

}
